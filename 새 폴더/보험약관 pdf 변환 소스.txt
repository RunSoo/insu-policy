import fitz  # PyMuPDF
import re

def to_markdown_table(table_data):
    """추출된 리스트 데이터를 마크다운 표 형식으로 변환"""
    if not table_data:
        return ""
    
    md_output = "\n"
    for i, row in enumerate(table_data):
        clean_row = [str(cell).replace('\n', ' ').strip() if cell else "" for cell in row]
        md_output += "| " + " | ".join(clean_row) + " |\n"
        if i == 0:  # 헤더 구분선
            md_output += "| " + " | ".join(["---"] * len(row)) + " |\n"
    return md_output + "\n"

def extract_all_content(pdf_path):
    doc = fitz.open(pdf_path)
    final_output = []

    for page in doc:
        # 1. 해당 페이지의 모든 표 찾기
        tables = page.find_tables()
        table_bboxes = [t.bbox for t in tables]
        
        # 2. 텍스트 블록 가져오기 (x0, y0, x1, y1, "text", block_no, block_type)
        blocks = page.get_text("blocks")
        # Y좌표(상하) -> X좌표(좌우) 순으로 정렬
        blocks.sort(key=lambda b: (b[1], b[0]))

        processed_table_indices = set()

        for b in blocks:
            block_rect = fitz.Rect(b[:4])
            is_inside_any_table = False
            
            for i, tab_bbox in enumerate(table_bboxes):
                # 텍스트 블록이 표 영역과 겹치는지 확인
                if block_rect.intersects(tab_bbox):
                    is_inside_any_table = True
                    # 해당 표를 아직 처리하지 않았다면 마크다운으로 변환하여 추가
                    if i not in processed_table_indices:
                        md_tab = to_markdown_table(tables[i].extract())
                        final_output.append(md_tab)
                        processed_table_indices.add(i)
                    break
            
            # 표 영역 밖의 일반 텍스트만 처리
            if not is_inside_any_table:
                text = b[4].strip()
                if text:
                    # Dify 청킹용 '제n조' 헤더 처리
                    text = re.sub(r"(제\s?\d+\s?조\(.*?\))", r"\n\n# \1", text)
                    final_output.append(text)

    return "\n\n".join(final_output)

# 실행 및 저장
result_text = extract_all_content("2.약관_신한 SOL 처음해외여행보험(20260316).pdf")
with open("integrated_knowledge.md", "w", encoding="utf-8") as f:
    f.write(result_text)
